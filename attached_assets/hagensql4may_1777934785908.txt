  embedding USER-DEFINED,
  fingerprint jsonb,
  source text DEFAULT 'manual'::text CHECK (source = ANY (ARRAY['manual'::text, 'ai'::text, 'migration'::text])),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT video_signals_pkey PRIMARY KEY (id),
  CONSTRAINT video_signals_video_id_fkey FOREIGN KEY (video_id) REFERENCES public.analyzed_videos(id),
  CONSTRAINT video_signals_brand_id_fkey FOREIGN KEY (brand_id) REFERENCES public.brand_profiles(id)
);
